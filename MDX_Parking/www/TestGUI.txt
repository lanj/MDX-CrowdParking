      initial-scale=1.0, maximum-scale=1.0,

    <script type="text/javascript" charset="utf-8"
      src="phonegap.js"></script>			
    <script type="text/javascript" charset="utf-8">
	  
	  
	  var accelero;
	  
	  var pr = false;
	  
	  function onBodyLoad() {
	    	document.addEventListener("deviceready",onDeviceReady,false);
	}  
	  

	function onDeviceReady(){
		
		acceleraro = document.getElementById('accelReport');
		
		//set the variable to let others know of phonegap initilization
		pr = true;
	}
	
	function getAcceleration(){
		if(pr==true){
			acceelero.innerHTML="";
			navigator.accelerometer.getCurrentAcceleration(onAccelSuccess, onAccelFailure);
			
		}else{
			alert("Please wait,\n PhoneGap is not ready");
		}
	}	
	
	function onAccelSuccess(accel){
		var d = new Date(accel.timestamp);
		accelero.innerHTML = "<b> Current acceleration </b><hr/>
		 X: " + accelero.x +
		"<br/>Y:" + accelero.y + 
		"<br/>Z:" + accelero.z + "<br/>
		Timestamp: " + d.toLocalString() +
		"<hr/>Click the button to refresh.";
	}
	function onAccelFailure(){
		alert("Accelerometer error!");
	}
	</script>
